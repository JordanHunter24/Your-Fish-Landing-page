#!/usr/bin/env python3 """ Clean up Your Fish Landing page index.html. Replaces all giant base64-encoded SVG logo blobs with references to the uploaded image file 'Your Fish Logo.png'. Reads from index.html in the same folder as this script and writes index_clean.html alongside it. """ import os import re import sys HERE = os.path.dirname(os.path.abspath(__file__)) INPUT = os.path.join(HERE, "index.html") OUTPUT = os.path.join(HERE, "index_clean.html") def main(): if not os.path.exists(INPUT): print("ERROR: Could not find index.html next to this script.") print(f"Expected at: {INPUT}") print("Put index.html in the same folder as this script and try again.") input("\nPress Enter to close...") sys.exit(1) with open(INPUT, "r", encoding="utf-8") as f: code = f.read() before = len(code) print(f"Read index.html — {before:,} characters ({before/1024/1024:.2f} MB)") # Replace every src="data:image/svg+xml;base64,...." with the PNG reference. # The %20 is URL-encoded space, since the uploaded filename has spaces. pattern = r'src="data:image/svg\+xml;base64,[A-Za-z0-9+/=]+"' replacement = 'src="Your%20Fish%20Logo.png"' new_code, count = re.subn(pattern, replacement, code) after = len(new_code) print(f"Replaced {count} base64 logo blob(s)") print(f"New size: {after:,} characters ({after/1024:.1f} KB)") print(f"Saved: {(before-after)/1024:.0f} KB") with open(OUTPUT, "w", encoding="utf-8") as f: f.write(new_code) print(f"\n✅ Done. Wrote: {OUTPUT}") print("\nNext step: upload index_clean.html to GitHub, but RENAME it to index.html when uploading.") input("\nPress Enter to close...") if __name__ == "__main__": main()
